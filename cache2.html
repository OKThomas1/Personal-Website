/**
* @param {string[]} equations
* @return {boolean}
*/
var equationsPossible = function(equations) {
let equal = equations.filter(e => e.charAt(1) === "=")
let notEqual = equations.filter(e => e.charAt(1) !== "=")
let map = {}
for(let i = 0; i < equal.length; i++) {
let e = equal[i]
if(!map[e.charAt(0)]) map[e.charAt(0)] = []
if(!map[e.charAt(3)]) map[e.charAt(3)] = []
map[e.charAt(0)].push(e.charAt(3))
map[e.charAt(3)].push(e.charAt(0))
}
let visited = {}
let i = "a".charCodeAt(0)
let num = 1
while(i <= "z".charCodeAt(0)) {
let queue = [String.fromCharCode(i)]
while(queue.length > 0) {
let letter = queue.shift()
if(!map?.[letter]) continue
if(visited[letter]) continue
for(let j = 0; j < map[letter].length; j++) {
queue.push(map[letter][j])
}
visited[letter] = num
}
i++
num++
}
for(let j = 0; j < notEqual.length; j++) {
let e = notEqual[j]
if(e.charAt(0) === e.charAt(3)) return false
let a = visited[e.charAt(0)] || Math.random()
let b = visited[e.charAt(3)] || Math.random()
if(a === b) return false
}
return true
};
